    <title>CloudKit Events</title>

    <script src="https://cdn.apple-cloudkit.com/ck/2/cloudkit.js"></script>

</head>

<body>

    <div id="apple-sign-in-button"></div>

    <div id="apple-sign-out-button"></div>

    <h1>Events</h1>

    <div id="events"></div>


    <script>

        CloudKit.configure({

            containers: [{

                containerIdentifier: 'iCloud.com.yourcompany.yourapp', // Replace with your container ID

                apiTokenAuth: {

                    apiToken: 'your-web-api-token', // Replace with your Web API token

                    persist: true,

                    signInButton: {

                        id: 'apple-sign-in-button',

                        theme: 'black' // Other options: 'white', 'white-with-outline'

                    },

                    signOutButton: {

                        id: 'apple-sign-out-button',

                        theme: 'black'

                    }

                },

                environment: 'production' // or 'development'

            }]

        });


        const container = CloudKit.getDefaultContainer();

        const publicDB = container.publicCloudDatabase;


        function fetchEvents() {

            const query = {

                recordType: 'Event',

                sortBy: [{

                    fieldName: 'timestamp',

                    ascending: false

                }]

            };


            publicDB.performQuery(query)

                .then(response => {

                    if (response.hasErrors) {

                        console.error(response.errors[0]);

                        return;

                    }


                    const events = response.records.map(record => {

                        return {

                            timestamp: record.fields.timestamp.value,

                            eventDescription: record.fields.eventDescription.value

                        };

                    });


                    displayEvents(events);

                });

        }


        function displayEvents(events) {

            const eventsContainer = document.getElementById('events');

            eventsContainer.innerHTML = '';


            events.forEach(event => {

                const eventElement = document.createElement('div');

                eventElement.innerHTML = `

                    <h3>${event.eventDescription}</h3>

                    <p>${new Date(event.timestamp).toLocaleString()}</p>

                `;

                eventsContainer.appendChild(eventElement);

            });

        }


        // Fetch events when the page loads

        window.onload = fetchEvents;

    </script>
